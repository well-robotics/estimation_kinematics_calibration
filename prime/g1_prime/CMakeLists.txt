cmake_minimum_required(VERSION 3.16)
project(g1_covariance_calibration NONE)

# Superbuild: first the vendored PRIME (libcrocoddyl.so), then the overlay
# (lower-solver CLI, pybind module, C++ tests).

include(ExternalProject)

set(PRIME_SOURCE ${CMAKE_CURRENT_SOURCE_DIR}/third_party/PRIME)
set(PRIME_BINARY ${CMAKE_CURRENT_BINARY_DIR}/prime)
set(OVERLAY_SOURCE ${CMAKE_CURRENT_SOURCE_DIR}/cpp)
set(OVERLAY_BINARY ${CMAKE_CURRENT_BINARY_DIR}/overlay)

ExternalProject_Add(prime
  SOURCE_DIR ${PRIME_SOURCE}
  BINARY_DIR ${PRIME_BINARY}
  CMAKE_ARGS
    -DCMAKE_BUILD_TYPE=Release
    -DBUILD_PYTHON_INTERFACE=OFF
    -DBUILD_BENCHMARK=OFF
    -DBUILD_EXAMPLES=OFF
    -DBUILD_TESTING=OFF
    -DBUILD_CONTACT_ID_EXPERIMENTS=OFF
    -DBUILD_WITH_IPOPT=OFF
    -DBUILD_WITH_MULTITHREADS=ON
    -DBUILD_WITH_NTHREADS=8
    -DINSTALL_DOCUMENTATION=OFF
  BUILD_COMMAND ${CMAKE_COMMAND} --build ${PRIME_BINARY} -j
  INSTALL_COMMAND ""
  BUILD_BYPRODUCTS ${PRIME_BINARY}/libcrocoddyl.so)

ExternalProject_Add(overlay
  DEPENDS prime
  SOURCE_DIR ${OVERLAY_SOURCE}
  BINARY_DIR ${OVERLAY_BINARY}
  CMAKE_ARGS
    -DCMAKE_BUILD_TYPE=Release
    -DPRIME_SRC=${PRIME_SOURCE}
    -DPRIME_BUILD=${PRIME_BINARY}
  BUILD_COMMAND ${CMAKE_COMMAND} --build ${OVERLAY_BINARY} -j
  INSTALL_COMMAND ""
  BUILD_ALWAYS ON)
